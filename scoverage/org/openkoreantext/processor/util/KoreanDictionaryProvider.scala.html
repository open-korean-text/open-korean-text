34   private[this] def readStreamByLine(stream: InputStream, filename: String): Iterator[String] = {
35     require(stream != null, "Resource not loaded: " + filename)
36     Source.fromInputStream(stream)(io.Codec("UTF-8"))
37         .getLines()
38         .map(_.trim)
39         .filter(_.length > 0)
40   }

42   private[this] def readWordFreqs(filename: String): collection.mutable.Map[CharSequence, Float] = {
43     var freqMap: collection.mutable.Map[CharSequence, Float] =
44       new java.util.HashMap[CharSequence, Float]
45 
46     readFileByLineFromResources(filename).foreach {
47       case line => if (line.contains("\t")) {
48         val data = line.split("\t")
49         freqMap += (data(0) -> data(1).slice(0, 6).toFloat)
50       }
51     }
52     freqMap
53   }
54 
55   private[this] def readWordMap(filename: String): Map[String, String] = {
56     readFileByLineFromResources(filename).filter {
57       case line: String => line.contains(" ")
58     }.map {
59       case line =>
60         val data = line.split(" ")
61         (data(0), data(1))
62     }.toMap
63   }

66   protected[processor] def readWordsAsSeq(filename: String): Seq[String] =

70   protected[processor] def readWordsAsSet(filenames: String*): Set[String] = {
71     filenames.foldLeft(Set[String]()) {

79   protected[processor] def readWords(filenames: String*): CharArraySet = {
80     val set = newCharArraySet
81     filenames.foreach(
82       filename => readFileByLineFromResources(filename).foreach(set.add)

84     set

87   protected[processor] def readFileByLineFromResources(filename: String): Iterator[String] = {
88     readStreamByLine(
89       if (filename.endsWith(".gz")) {
90         new GZIPInputStream(getClass.getResourceAsStream(filename))
91       } else {
92         getClass.getResourceAsStream(filename)
93       }
94       , filename
95     )
96   }
