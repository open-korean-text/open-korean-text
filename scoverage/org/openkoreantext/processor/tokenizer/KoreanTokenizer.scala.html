33 /**
34   * Provides Korean tokenization.
35   *
36   * Chunk: 어절 - 공백으로 구분되어 있는 단위 (사랑하는사람을)
37   * Word: 단어 - 하나의 문장 구성 요소 (사랑하는, 사람을)
38   * Token: 토큰 - 형태소와 비슷한 단위이지만 문법적으로 정확하지는 않음 (사랑, 하는, 사람, 을)
39   *
40   * Whenever there is an updates in the behavior of KoreanParser,
41   * the initial cache has to be updated by running tools.CreateInitialCache.
42   */

44   private val TOP_N_PER_STATE = 5

46   /**
47     * 0 for optional, 1 for required
48     * * for optional repeatable, + for required repeatable
49     *
50     * Substantive: 체언 (초거대기업의)
51     * Predicate: 용언 (하였었습니다, 개예뻤었다)
52     * Modifier: 수식언 (모르는 할수도있는 보이기도하는 예뻐 예쁜 완전 레알 초인간적인 잘 잘한)
53     * Standalone: 독립언
54     * Functional: 관계언 (조사)
55     *
56     * N Noun: 명사 (Nouns, Pronouns, Company Names, Proper Noun, Person Names, Numerals, Standalone, Dependent)
57     * V Verb: 동사 (하, 먹, 자, 차)
58     * J Adjective: 형용사 (예쁘다, 크다, 작다)
59     * A Adverb: 부사 (잘, 매우, 빨리, 반드시, 과연)
60     * D Determiner: 관형사 (새, 헌, 참, 첫, 이, 그, 저)
61     * E Exclamation: 감탄사 (헐, ㅋㅋㅋ, 어머나, 얼씨구)
62     *
63     * C Conjunction: 접속사
64     *
65     * j SubstantiveJosa: 조사 (의, 에, 에서)
66     * l AdverbialJosa: 부사격 조사 (~인, ~의, ~일)
67     * e Eomi: 어말어미 (다, 요, 여, 하댘ㅋㅋ)
68     * r PreEomi: 선어말어미 (었)
69     *
70     * m Modifier: 관형사 ('초'대박)
71     * v VerbPrefix: 동사 접두어 ('쳐'먹어)
72     * s Suffix: 접미사 (~적)
73     */

102   /**
103     * Parse Korean text into a sequence of KoreanTokens with custom parameters
104     *
105     * @param text Input Korean chunk
106     * @return sequence of KoreanTokens
107     */
108   def tokenizeTopN(text: CharSequence,
109                    topN: Int = 1,
110                    profile: TokenizerProfile = TokenizerProfile.defaultProfile
111                   ): Seq[Seq[Seq[KoreanToken]]] = {
112     try
113       chunk(text).map {
114         case token: KoreanToken if token.pos == Korean =>
115           // Get the best parse of each chunk
116           val parsed = parseKoreanChunk(token, profile, topN)
117 
118           // Collapse sequence of one-char nouns into one unknown noun: (가Noun 회Noun -> 가회Noun*)
119           parsed.map(collapseNouns)
120         case token: KoreanToken => Seq(Seq(token))
121       }
122     catch {
123       case e: Exception =>
124         System.err.println(s"Error tokenizing a chunk: $text")
125         throw e
126     }
127   }

129   /**
130     * Find the best parse using dynamic programming.
131     *
132     * @param chunk Input chunk. The input has to be entirely. Check for input validity is skipped
133     *              for performance optimization. This method is private and is called only by tokenize.
134     * @return The best possible parse.
135     */
136   private[this] def parseKoreanChunk(chunk: KoreanToken,
137                                      profile: TokenizerProfile = TokenizerProfile.defaultProfile,
138                                      topN: Int = 1
139                                     ): Seq[Seq[KoreanToken]] = {
140 
141     val candidates = findTopCandidates(chunk, profile)
142     candidates.take(topN)
143   }

145   private def findTopCandidates(chunk: KoreanToken, profile: TokenizerProfile) = {
146     val directMatch: Seq[Seq[KoreanToken]] = findDirectMatch(chunk)
147 
148     // Buffer for solutions
149     val solutions: util.HashMap[Int, List[CandidateParse]] = new java.util.HashMap[Int, List[CandidateParse]]
150 
151     // Initial state
152     solutions.put(0, List(
153       CandidateParse(
154         ParsedChunk(Seq[KoreanToken](), 1, profile),
155         koreanPosTrie, ending = None
156       )
157     ))
158 
159     // Find N best parses per state
160     for (
161       end <- 1 to chunk.length;
162       start <- end - 1 to(Seq(end - MAX_TRACE_BACK, 0).max, -1)
163     ) {
164       val word = chunk.text.slice(start, end)
165 
166       val curSolutions = solutions.get(start)
167 
168       val candidates = curSolutions.flatMap {
169         solution =>
170           val possiblePoses: Seq[PossibleTrie] = if (solution.ending.isDefined) {
171             solution.curTrie.map(t => PossibleTrie(t, 0)) ++ koreanPosTrie.map(
172               t => PossibleTrie(t, 1))
173           } else {
174             solution.curTrie.map(t => PossibleTrie(t, 0))
175           }
176 
177           possiblePoses.view.filter { t =>
178             t.curTrie.curPos == Noun || koreanDictionary.get(t.curTrie.curPos).contains(
179               word.toCharArray)
180           }.map { t: PossibleTrie =>
181             val candidateToAdd =
182               if (t.curTrie.curPos == Noun && !koreanDictionary.get(Noun).contains(word.toCharArray)) {
183                 val isWordName: Boolean = isName(word)
184                 val isWordKoreanNameVariation: Boolean = isKoreanNameVariation(word)
185 
186                 val unknown = !isWordName && !isKoreanNumber(word) && !isWordKoreanNameVariation
187                 val pos = Noun
188                 ParsedChunk(Seq(KoreanToken(word, pos, chunk.offset + start, word.length, unknown = unknown)),
189                   t.words, profile)
190               } else {
191                 val pos = t.curTrie.curPos
192                 ParsedChunk(Seq(KoreanToken(word, pos, chunk.offset + start, word.length)), t.words,
193                   profile)
194               }
195 
196             val nextTrie = t.curTrie.nextTrie.map {
197               case nt: KoreanPosTrie if nt == selfNode => t.curTrie
198               case nt: KoreanPosTrie => nt
199             }
200 
201             CandidateParse(solution.parse ++ candidateToAdd, nextTrie, t.curTrie.ending)
202           }
203       }
204 
205       val currentSolutions = if (solutions.containsKey(end)) solutions.get(end) else List()
206 
207       solutions.put(end, (currentSolutions ++ candidates).sortBy {
208         c => (c.parse.score, c.parse.posTieBreaker)
209       }.take(TOP_N_PER_STATE))
210     }
211 
212 
213     val topCandidates: Seq[Seq[KoreanToken]] = if (solutions.get(chunk.length).isEmpty) {
214       // If the chunk is not parseable, treat it as a unknown noun chunk.
215       Seq(Seq(KoreanToken(chunk.text, Noun, 0, chunk.length, unknown = true)))
216     } else {
217       // Return the best parse of the final state
218       solutions.get(chunk.length).sortBy(c => c.parse.score).map { p => p.parse.posNodes }
219     }
220 
221     (directMatch ++ topCandidates).distinct
222   }

224   private def findDirectMatch(chunk: KoreanToken): Seq[Seq[KoreanToken]] = {
225     // Direct match
226     // This may produce 하 -> PreEomi
227     koreanDictionary.asScala.foreach {
228       case (pos, dict) =>
229         if (dict.contains(chunk.text)) {
230           return Seq(Seq(KoreanToken(chunk.text, pos, chunk.offset, chunk.length)))
231         }
232     }
233     Seq()
234   }

236   case class KoreanToken(text: String, pos: KoreanPos, offset: Int, length: Int,
237                          stem: Option[String] = None, unknown: Boolean = false) {
238     override def toString: String = {
239       val unknownStar = if (unknown) "*" else ""
240       val stemString = if (stem.isDefined) "(%s)".format(stem.get) else ""
241       s"$text$unknownStar(${pos.toString}$stemString: $offset, $length)"
242     }
243 
244     def copyWithNewPos(pos: KoreanPos): KoreanToken = {
245       KoreanToken(this.text, pos, this.offset, this.length, unknown = this.unknown)
246     }
247   }
