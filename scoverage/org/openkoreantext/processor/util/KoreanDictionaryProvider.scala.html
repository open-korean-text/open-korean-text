43   private[this] def readWordFreqs(filename: String): util.HashMap[CharSequence, Float] = {
44     var freqMap: util.HashMap[CharSequence, Float] =
45       new java.util.HashMap[CharSequence, Float]
46 
47     readFileByLineFromResources(filename).foreach {
48       case line => if (line.contains("\t")) {
49         val data = line.split("\t")

52     }

56   private[this] def readWordMap(filename: String): Map[String, String] = {
57     readFileByLineFromResources(filename).filter {
58       case line: String => line.contains(" ")
59     }.map {
60       case line =>
61         val data = line.split(" ")
62         (data(0), data(1))
63     }.toMap
64   }

71   protected[processor] def readWordsAsSet(filenames: String*): Set[String] = {

80   protected[processor] def readWords(filenames: String*): CharArraySet = {

88   protected[processor] def readFileByLineFromResources(filename: String): Iterator[String] = {
89     readStreamByLine(
90       if (filename.endsWith(".gz")) {

92       } else {
93         getClass.getResourceAsStream(filename)
94       }
95       , filename
96     )
97   }
