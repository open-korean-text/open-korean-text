83     def trimSpacesFromPhrase(phrases: Seq[KoreanPhrase]): Seq[KoreanPhrase] = {
84       phrases.zipWithIndex.map {
85         case (phrase, i) if phrases.length == 1 =>
86           KoreanPhrase(phrase.tokens
87             .dropWhile(_.pos == Space)
88             .reverse.dropWhile(_.pos == Space).reverse, phrase.pos)
89         case (phrase, i) if i == 0 =>
90           KoreanPhrase(phrase.tokens.dropWhile(_.pos == Space), phrase.pos)
91         case (phrase, i) if i == phrases.length - 1 =>
92           KoreanPhrase(phrase.tokens.reverse.dropWhile(_.pos == Space).reverse, phrase.pos)
93         case (phrase, i) => phrase
94       }
95     }

107     def notEndingInNonPhraseSuffix: Boolean = {
108       val lastToken = phraseChunk.last.tokens.last
109       !(lastToken.pos == Suffix && lastToken.text == "적")

111 
112     def isRightLength: Boolean = {
113       val phraseChunkWithoutSpaces: Seq[KoreanPhrase] = phraseChunk.filter(_.pos != Space)
114 
115       def checkMaxLength: Boolean = {
116         phraseChunkWithoutSpaces.length <= MaxPhrasesPerPhraseChunk &&
117           phraseChunkWithoutSpaces.map(_.length).sum <= MaxCharsPerPhraseChunkWithoutSpaces
118       }
119 
120       def checkMinLength: Boolean = {
121         phraseChunkWithoutSpaces.length >= MinPhrasesPerPhraseChunk ||
122           (phraseChunkWithoutSpaces.length < MinPhrasesPerPhraseChunk &&
123             phraseChunkWithoutSpaces.map(_.length).sum >= MinCharsPerPhraseChunkWithoutSpaces)
124       }
125 
126       def checkMinLengthPerToken: Boolean = {
127         phraseChunkWithoutSpaces.exists(_.length > 1)
128       }
129 
130       checkMaxLength && checkMinLength && checkMinLengthPerToken
131     }
132 
133     isRightLength && notEndingInNonPhraseSuffix
134   }

152   protected[processor] def collapsePos(tokens: Seq[KoreanToken]): Seq[KoreanPhrase] = {
153     def getTries(token: KoreanToken, trie: List[KoreanPosTrie]): (KoreanPosTrie, List[KoreanPosTrie]) = {
154       val curTrie = trie.filter(_.curPos == token.pos).head
155       val nextTrie = curTrie.nextTrie.map {
156         case nt: KoreanPosTrie if nt == selfNode => curTrie
157         case nt: KoreanPosTrie => nt
158       }
159       (curTrie, nextTrie)

162     def getInit(phraseBuffer: PhraseBuffer): List[KoreanPhrase] = {
163       if (phraseBuffer.phrases.isEmpty) {
164         List()
165       } else {
166         phraseBuffer.phrases.init
167       }
168     }
169 
170     tokens.foldLeft(PhraseBuffer(List[KoreanPhrase](), CollapseTrie, None)) {
171       case (output, token) if output.curTrie.exists(_.curPos == token.pos) =>
172         // Extend the current phrase
173         val (ct, nt) = getTries(token, output.curTrie)
174 
175         if (output.phrases.isEmpty || output.curTrie == CollapseTrie) {
176           PhraseBuffer(
177             output.phrases :+ KoreanPhrase(List(token), ct.ending.getOrElse(Noun)),
178             nt, ct.ending
179           )
180         } else {
181           PhraseBuffer(
182             getInit(output) :+
183               KoreanPhrase(output.phrases.last.tokens :+ token, ct.ending.getOrElse(Noun)),
184             nt, ct.ending
185           )
186         }
187       case (output, token) if CollapseTrie.exists(_.curPos == token.pos) =>
188         // Start a new phrase
189         val (ct, nt) = getTries(token, CollapseTrie)
190 
191         PhraseBuffer(
192           output.phrases :+ KoreanPhrase(List(token), ct.ending.getOrElse(Noun)),
193           nt, ct.ending
194         )
195       case (output, token) =>
196         // Add a single word
197         PhraseBuffer(
198           output.phrases :+ KoreanPhrase(List(token), token.pos),
199           CollapseTrie, output.ending
200         )
201     }.phrases
202   }

204   private def distinctPhrases(chunks: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
205     val (l, buffer) = chunks.foldLeft((List[KoreanPhraseChunk](), Set[String]())) {

224     def isNonNounPhraseCandidate(phrase: KoreanPhrase): Boolean = {
225       val trimmed = trimPhrase(phrase)
226 
227       // 하는, 할인된, 할인될, exclude: 하지만
228       def isModifyingPredicate: Boolean = {
229         val lastChar: Char = trimmed.tokens.last.text.last
230         (trimmed.pos == Verb || trimmed.pos == Adjective) &&
231           ModifyingPredicateEndings.contains(Hangul.decomposeHangul(lastChar).coda) &&
232           !ModifyingPredicateExceptions.contains(lastChar)
233       }
234 
235       // 과, 와, 의
236       def isConjuction: Boolean =
237         trimmed.pos == Josa && ConjunctionJosa.contains(trimmed.tokens.last.text)
238 
239       def isAlphaNumeric: Boolean =
240         trimmed.pos == Alpha || trimmed.pos == Number
241 
242       isAlphaNumeric || isModifyingPredicate || isConjuction
243     }

245     def collapseNounPhrases(phrases: KoreanPhraseChunk): KoreanPhraseChunk = {
246       val (output, buffer) = phrases.foldLeft((Seq[KoreanPhrase](), Seq[KoreanPhrase]())) {
247         case ((output, buffer), phrase) if phrase.pos == Noun || phrase.pos == ProperNoun =>
248           (output, buffer :+ phrase)
249         case ((output, buffer), phrase) =>
250           val tempPhrases = if (buffer.length > 0) {
251             Seq(KoreanPhrase(buffer.flatMap(_.tokens)), phrase)
252           } else {
253             Seq(phrase)
254           }
255           (output ++ tempPhrases, Seq[KoreanPhrase]())
256       }
257       if (buffer.length > 0) output :+ KoreanPhrase(buffer.flatMap(_.tokens)) else output
258     }

260     def collapsePhrases(phrases: KoreanPhraseChunk): Seq[KoreanPhraseChunk] = {
261       def addPhraseToBuffer(phrase: KoreanPhrase, buffer: Seq[KoreanPhraseChunk]): Seq[KoreanPhraseChunk] = {
262         buffer.map(b => b :+ phrase)
263       }
264 
265       def newBuffer: Seq[Seq[KoreanPhrase]] = Seq(Seq[KoreanPhrase]())
266 
267       val (output, buffer) = phrases.foldLeft((Seq[KoreanPhraseChunk](), newBuffer)) {
268         case ((output, buffer), phrase) if PhraseTokens.contains(phrase.pos) && isNotSpam(phrase) =>
269           val bufferWithThisPhrase = addPhraseToBuffer(phrase, buffer)
270           if (phrase.pos == Noun || phrase.pos == ProperNoun) {
271             (output ++ bufferWithThisPhrase, bufferWithThisPhrase)
272           } else {
273             (output, bufferWithThisPhrase)
274           }
275         case ((output, buffer), phrase) if isNonNounPhraseCandidate(phrase) =>
276           (output, addPhraseToBuffer(phrase, buffer))
277         case ((output, buffer), phrase) =>
278           (output ++ buffer, newBuffer)
279       }
280       if (buffer.length > 0) output ++ buffer else output
281     }

283     def getSingleTokenNouns: Seq[KoreanPhraseChunk] = {
284       phrases.filter {

299   /**
300    * Find suitable phrases
301    *
302    * @param tokens A sequence of tokens
303    * @param filterSpam true if spam words and slangs to be filtered out
304    * @param addHashtags true if #hashtags to be included
305    * @return A list of KoreanPhrase
306    */
307   def extractPhrases(tokens: Seq[KoreanToken],
308                      filterSpam: Boolean = false,
309                      addHashtags: Boolean = true): Seq[KoreanPhrase] = {
310     val hashtags = tokens.flatMap {
311       case t: KoreanToken if t.pos == KoreanPos.Hashtag => Some(KoreanPhrase(Seq(t), KoreanPos.Hashtag))
312       case t: KoreanToken if t.pos == KoreanPos.CashTag => Some(KoreanPhrase(Seq(t), KoreanPos.CashTag))
313       case _ => None
314     }
315 
316     val collapsed = collapsePos(tokens)
317     val candidates = getCandidatePhraseChunks(collapsed, filterSpam)
318     val permutatedCandidates = permutateCadidates(candidates)
319 
320     val phrases = permutatedCandidates.map {
321       phraseChunk: KoreanPhraseChunk => KoreanPhrase(trimPhraseChunk(phraseChunk).flatMap(_.tokens))
322     }
323 
324     if (addHashtags) {
325       phrases ++ hashtags
326     } else {
327       phrases
328     }
329   }
