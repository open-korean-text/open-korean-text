33  * Split input text into Korean Chunks (어절)
34  */
35 object KoreanChunker {
36   private[tokenizer] val POS_PATTERNS = Map(
37     Korean -> """([가-힣]+)""".r.pattern,
38     Alpha -> """(\p{Alpha}+)""".r.pattern,
39     Number -> ("""(\$?\p{Digit}+"""
40       + """(,\p{Digit}{3})*([/~:\.-]\p{Digit}+)?"""
41       + """(천|만|억|조)*(%|원|달러|위안|옌|엔|유로|등|년|월|일|회|시간|시|분|초)?)""").r.pattern,
42     KoreanParticle -> """([ㄱ-ㅣ]+)""".r.pattern,
43     Punctuation -> """([\p{Punct}·…’]+)""".r.pattern,
44     URL -> Regex.VALID_URL,
45     Email -> """([\p{Alnum}\.\-_]+@[\p{Alnum}\.]+)""".r.pattern,
46     Hashtag -> Regex.VALID_HASHTAG,
47     ScreenName -> Regex.VALID_MENTION_OR_LIST,
48     CashTag -> Regex.VALID_CASHTAG,
49     Space -> """\s+""".r.pattern
50   )
51 
52   private val CHUNKING_ORDER = Seq(URL, Email, ScreenName, Hashtag, CashTag, Number, Korean, KoreanParticle, Alpha, Punctuation)
53 
54   protected[processor] def getChunks(input: String, keepSpace: Boolean = false): Seq[String] = {
55     chunk(input).map(_.text)
56   }
57 
58   private[tokenizer] case class ChunkMatch(start: Int, end: Int, text: String, pos: KoreanPos) {
59     def disjoint(that: ChunkMatch): Boolean = {
60       (that.start < this.start && that.end <= this.start) ||
61         (that.start >= this.end && that.end > this.end)
62     }
63   }
64 
65   def splitBySpaceKeepingSpace(s: CharSequence): Seq[String] = {
66     val space = """\s+""".r.pattern
67     val m = space.matcher(s)
68 
69     var tokens = new ListBuffer[String]()
70     var index = 0
71     while(m.find()) {
72       if (index < m.start) {
73         tokens += s.subSequence(index, m.start).toString
74       }
75       tokens += s.subSequence(m.start, m.end).toString
76       index = m.end
77     }
78 
79     if (index < s.length()) {
80       tokens += s.subSequence(index, s.length()).toString
81     }
82 
83     return tokens.toList
84   }
85 
86   /**
87    * Recursively call m.find() to find all the matches.
88    * Use tail-recursion optimization to avoid stack overflow.
89    *
90    * @param m input Matcher
91    * @param pos KoreanPos to attach
92    * @param matches ouput list of ChunkMatch
93    * @return list of ChunkMatches
94    */
95   @scala.annotation.tailrec
96   private[tokenizer] def findAllPatterns(m: Matcher, pos: KoreanPos, matches: List[ChunkMatch] = List()): List[ChunkMatch] = {
97     if (m.find()) {
98       findAllPatterns(m, pos, ChunkMatch(m.start, m.end, m.group(), pos) :: matches)
99     } else {
100       matches
101     }
102   }
103 
104   private[this] def splitChunks(text: String): List[ChunkMatch] = {
105     if (text.charAt(0).isSpaceChar) {
106       List(ChunkMatch(0, text.length, text, Space))
107     } else {
108       val chunksBuf = new ListBuffer[ChunkMatch]()
109       var matchedLen = 0
110       CHUNKING_ORDER.foreach { pos =>
111         if (matchedLen < text.length) {
112           val m = POS_PATTERNS(pos).matcher(text)
113           while (m.find()) {
114             val cm = ChunkMatch(m.start, m.end, m.group(), pos)
115             if (chunksBuf.forall(cm.disjoint)) {
116               chunksBuf += cm
117               matchedLen += cm.end - cm.start
118             }
119           }
120         }
121       }
122 
123       val chunks = chunksBuf.sortBy(cm => cm.start).toList
124       fillInUnmatched(text, chunks, Foreign)
125     }
126   }
127 
128   /**
129    * Fill in unmatched segments with given pos
130    *
131    * @param text input text
132    * @param chunks matched chunks
133    * @param pos KoreanPos to attach to the unmatched chunk
134    * @return list of ChunkMatches
135    */
136   private[this] def fillInUnmatched(text: String,
137                                     chunks: Seq[ChunkMatch],
138                                     pos: KoreanPos.Value): List[ChunkMatch] = {
139 
140     // Add Foreign for unmatched parts
141     val (chunksWithForeign, prevEnd) = chunks.foldLeft((List[ChunkMatch](), 0)) {
142       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start == prevEnd =>
143         (cm :: l, cm.end)
144       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) if cm.start > prevEnd =>
145         (cm :: ChunkMatch(prevEnd, cm.start, text.slice(prevEnd, cm.start), pos) :: l, cm.end)
146       case ((l: List[ChunkMatch], prevEnd: Int), cm: ChunkMatch) =>
147         throw new IllegalStateException("Non-disjoint chunk matches found.")
148     }
149 
150     val output = if (prevEnd < text.length) {
151       ChunkMatch(prevEnd, text.length, text.slice(prevEnd, text.length), pos) :: chunksWithForeign
152     } else {
153       chunksWithForeign
154     }
155 
156     output.reverse
157   }
158 
159 
160   /**
161    * Get chunks by given pos.
162    *
163    * @param input input string
164    * @param pos one of supported KoreanPos's: URL, Email, ScreenName, Hashtag,
165    *            CashTag, Korean, KoreanParticle, Number, Alpha, Punctuation
166    * @return sequence of Korean chunk strings
167    */
168   def getChunksByPos(input: String, pos: KoreanPos): Seq[KoreanToken] = {
169     chunk(input).filter(_.pos == pos)
170   }
171 
172   /**
173    * Split input text into a sequnce of KoreanToken. A candidate for Korean parser
174    * gets tagged with KoreanPos.Korean.
175    *
176    * @param input input string
177    * @return sequence of KoreanTokens
178    */
179   def chunk(input: CharSequence): Seq[KoreanToken] = {
180     val s = input.toString
181     val (l: List[KoreanToken], i: Int) = splitBySpaceKeepingSpace(s).flatMap {
182       s => splitChunks(s)
183     }.foldLeft(List[KoreanToken](), 0) {
184       case ((l: List[KoreanToken], i: Int), m: ChunkMatch) =>
185         val segStart = s.indexOf(m.text, i)
186         (KoreanToken(m.text, m.pos, segStart, m.text.length) :: l, segStart + m.text.length)
187     }
188     l.reverse
189   }
