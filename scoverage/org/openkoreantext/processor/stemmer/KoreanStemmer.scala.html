11   private val Endings = Set(Eomi, PreEomi)

16   /**
17    * Removes Ending tokens recovering the root form of predicates
18    *
19    * @param tokens A sequence of tokens
20    * @return A sequence of collapsed Korean tokens
21    */
22   def stem(tokens: Seq[KoreanToken]): Seq[KoreanToken] = {
23     if (!tokens.exists(t => t.pos == Verb || t.pos == Adjective)) {
24       return tokens
25     }
26 
27     tokens.foldLeft(List[KoreanToken]()) {
28       case (l: List[KoreanToken], token: KoreanToken) if l.nonEmpty && Endings.contains(token.pos) =>
29         if (Predicates.contains(l.head.pos)) {
30           val prevToken = l.head
31           KoreanToken(
32             prevToken.text + token.text,
33             prevToken.pos, prevToken.offset, prevToken.length + token.length,
34             stem = prevToken.stem,
35             unknown = prevToken.unknown
36           ) :: l.tail
37         } else {
38           token :: l
39         }
40       case (l: List[KoreanToken], token: KoreanToken) if Predicates.contains(token.pos) =>
41         KoreanToken(
42           token.text,
43           token.pos, token.offset, token.length,
44           stem = Some(predicateStems(token.pos)(token.text)),
45           unknown = token.unknown
46         ) :: l
47       case (l: List[KoreanToken], token: KoreanToken) => token :: l
48     }.reverse
49   }
