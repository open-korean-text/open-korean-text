26 /**
27  * Expands Korean verbs and adjectives to all possible conjugation forms.
28  */
29 object KoreanConjugation {
30   // ㅋ, ㅎ for 잨ㅋㅋㅋㅋ 잔댛ㅎㅎㅎㅎ
31   private[this] val CODAS_COMMON = Seq('ㅂ', 'ㅆ', 'ㄹ', 'ㄴ', 'ㅁ')
32   // 파랗다 -> 파래, 파램, 파랠, 파랬
33   private[this] val CODAS_FOR_CONTRACTION = Seq('ㅆ', 'ㄹ', 'ㅁ')
34   private[this] val CODAS_NO_PAST = Seq('ㅂ', 'ㄹ', 'ㄴ', 'ㅁ')
35 
36   private[this] val CODAS_SLANG_CONSONANT = Seq('ㅋ', 'ㅎ')
37   private[this] val CODAS_SLANG_VOWEL = Seq('ㅜ', 'ㅠ')
38 
39   private[this] val PRE_EOMI_COMMON = "게겠고구기긴길네다더던도든면자잖재져죠지진질".toSeq
40   private[this] val PRE_EOMI_1_1 = "야서써도준".toSeq
41   private[this] val PRE_EOMI_1_2 = "어었".toSeq
42   private[this] val PRE_EOMI_1_3 = "아았".toSeq
43   private[this] val PRE_EOMI_1_4 = "워웠".toSeq
44   private[this] val PRE_EOMI_1_5 = "여였".toSeq
45 
46   private[this] val PRE_EOMI_2 = "노느니냐".toSeq
47   private[this] val PRE_EOMI_3 = "러려며".toSeq
48   private[this] val PRE_EOMI_4 = "으".toSeq
49   private[this] val PRE_EOMI_5 = "은".toSeq
50   private[this] val PRE_EOMI_6 = "는".toSeq
51   private[this] val PRE_EOMI_7 = "운".toSeq
52 
53   private[this] val PRE_EOMI_RESPECT = "세시실신셔습셨십".toSeq
54 
55   // 모음 어말어미
56   private[this] val PRE_EOMI_VOWEL = PRE_EOMI_COMMON ++ PRE_EOMI_2 ++ PRE_EOMI_3 ++ PRE_EOMI_RESPECT
57 
58   private[this] def addPreEomi(lastChar: Char, charsToAdd: Seq[Char]): Seq[String] = {
59     charsToAdd.map(lastChar + _.toString)
60   }
61 
62   /**
63    * Wrap conjugatePredicated
64    *
65    * @param words Set of adjectives or verbs.
66    * @param isAdjective True if the set contains adjectives.
67    * @return CharArraySet with conjugated words.
68    */
69   protected[processor] def conjugatePredicatesToCharArraySet(words: Set[String], isAdjective: Boolean = false): CharArraySet = {
70     val expanded: Set[String] = conjugatePredicated(words, isAdjective)
71 
72     val newSet = newCharArraySet
73     newSet.addAll(expanded.asJava)
74     newSet
75   }
76 
77   /**
78    * Conjugate adjectives and verbs.
79    *
80    * @param words Set of adjectives or verbs.
81    * @param isAdjective True if the set contains adjectives.
82    * @return Set of conjugated words
83    */
84   protected[processor] def conjugatePredicated(words: Set[String], isAdjective: Boolean): Set[String] = {
85     lazy val expanded = words.flatMap { word: String =>
86       val init = word.init
87       val lastChar = word.last
88       val lastCharString = lastChar.toString
89       val lastCharDecomposed = decomposeHangul(lastChar)
90 
91       val expandedLast: Seq[String] = lastCharDecomposed match {
92         // Cases without codas
93         // 하다, special case
94         case HangulChar('ㅎ', 'ㅏ', ' ') =>
95           val endings = if (isAdjective) Seq("합", "해", "히", "하") else Seq("합", "해")
96           addPreEomi(lastChar, PRE_EOMI_COMMON ++ PRE_EOMI_2 ++ PRE_EOMI_6 ++ PRE_EOMI_RESPECT) ++
97               CODAS_COMMON.map {
98                 case c: Char if c == 'ㅆ' => composeHangul('ㅎ', 'ㅐ', c).toString
99                 case c: Char => composeHangul('ㅎ', 'ㅏ', c).toString
100               } ++ addPreEomi('하', PRE_EOMI_VOWEL ++ PRE_EOMI_1_5 ++ PRE_EOMI_6) ++
101               addPreEomi('해', PRE_EOMI_1_1) ++ endings
102 
103         // 쏘다
104         case HangulChar(o: Char, 'ㅗ', ' ') =>
105           addPreEomi(lastChar, PRE_EOMI_VOWEL ++ PRE_EOMI_2 ++ PRE_EOMI_1_3 ++ PRE_EOMI_6) ++
106               CODAS_NO_PAST.map(composeHangul(o, 'ㅗ', _).toString) ++
107               Seq(composeHangul(o, 'ㅘ', ' ').toString,
108                 composeHangul(o, 'ㅘ', 'ㅆ').toString,
109                 lastCharString)
110 
111         // 맞추다, 겨누다, 재우다,
112         case HangulChar(o: Char, 'ㅜ', ' ') =>
113           addPreEomi(lastChar, PRE_EOMI_VOWEL ++ PRE_EOMI_1_2 ++ PRE_EOMI_2 ++ PRE_EOMI_6) ++
114               CODAS_NO_PAST.map(composeHangul(o, 'ㅜ', _).toString) ++
115               Seq(composeHangul(o, 'ㅝ').toString,
116                 composeHangul(o, 'ㅝ', 'ㅆ').toString,
117                 lastCharString)
118 
119         // 치르다, 구르다, 굴르다, 뜨다, 모으다, 고르다, 골르다
120         case HangulChar(o: Char, 'ㅡ', ' ') =>
121           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
122               CODAS_NO_PAST.map(composeHangul(o, 'ㅡ', _).toString) ++
123               Seq(composeHangul(o, 'ㅝ').toString,
124                 composeHangul(o, 'ㅓ').toString,
125                 composeHangul(o, 'ㅏ').toString,
126                 composeHangul(o, 'ㅝ', 'ㅆ').toString,
127                 composeHangul(o, 'ㅓ', 'ㅆ').toString,
128                 composeHangul(o, 'ㅏ', 'ㅆ').toString,
129                 lastCharString)
130 
131         // 사귀다
132         case HangulChar('ㄱ', 'ㅟ', ' ') =>
133           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
134               CODAS_NO_PAST.map(composeHangul('ㄱ', 'ㅟ', _).toString) ++
135               Seq(composeHangul('ㄱ', 'ㅕ', ' ').toString, composeHangul('ㄱ', 'ㅕ', 'ㅆ').toString) ++
136               Seq(lastCharString)
137 
138         // 쥐다
139         case HangulChar(o: Char, 'ㅟ', ' ') =>
140           CODAS_NO_PAST.map(composeHangul(o, 'ㅟ', _).toString) ++
141               addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
142               Seq(lastCharString)
143 
144         // 마시다, 엎드리다, 치다, 이다, 아니다
145         case HangulChar(o: Char, 'ㅣ', ' ') =>
146           CODAS_NO_PAST.map(composeHangul(o, 'ㅣ', _).toString) ++
147               addPreEomi(lastChar, PRE_EOMI_1_2 ++ PRE_EOMI_2 ++ PRE_EOMI_6) ++
148               Seq(composeHangul(o, 'ㅣ', 'ㅂ') + "니",
149                 composeHangul(o, 'ㅕ', ' ').toString,
150                 composeHangul(o, 'ㅕ', 'ㅆ').toString,
151                 lastCharString)
152 
153         // 꿰다, 꾀다
154         case HangulChar(o: Char, v: Char, ' ') if v == 'ㅞ' || v == 'ㅚ' || v == 'ㅙ' =>
155           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
156               CODAS_COMMON.map(composeHangul(o, v, _).toString) ++
157               Seq(lastCharString)
158 
159         // All other vowel endings: 둘러서다, 켜다, 세다, 캐다, 차다
160         case HangulChar(o: Char, v: Char, ' ') =>
161           CODAS_COMMON.map(composeHangul(o, v, _).toString) ++
162               addPreEomi(lastChar, PRE_EOMI_VOWEL ++ PRE_EOMI_1_1 ++ PRE_EOMI_2 ++ PRE_EOMI_6) ++
163               Seq(lastCharString)
164 
165         // Cases with codas
166         // 만들다, 알다, 풀다
167         case HangulChar(o: Char, v: Char, 'ㄹ') if (o == 'ㅁ' && v == 'ㅓ') || v == 'ㅡ' || v == 'ㅏ' || v == 'ㅜ' =>
168           addPreEomi(lastChar, PRE_EOMI_1_2 ++ PRE_EOMI_3) ++
169               addPreEomi(composeHangul(o, v, ' '),
170                 PRE_EOMI_2 ++ PRE_EOMI_6 ++ PRE_EOMI_RESPECT) ++
171               Seq(composeHangul(o, v, 'ㄻ').toString,
172                 composeHangul(o, v, 'ㄴ').toString,
173                 lastCharString)
174 
175         // 낫다, 뺴앗다
176         case HangulChar(o: Char, 'ㅏ', 'ㅅ') =>
177           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
178               addPreEomi(composeHangul(o, 'ㅏ'), PRE_EOMI_4 ++ PRE_EOMI_5) ++
179               Seq(lastCharString)
180 
181         // 묻다
182         case HangulChar('ㅁ', 'ㅜ', 'ㄷ') =>
183           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
184               Seq(composeHangul('ㅁ', 'ㅜ', 'ㄹ').toString,
185                 lastCharString)
186 
187         // 붇다
188         case HangulChar(o: Char, 'ㅜ', 'ㄷ') =>
189           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
190               addPreEomi(composeHangul(o, 'ㅜ', ' '),
191                 PRE_EOMI_1_2 ++ PRE_EOMI_1_4 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
192               Seq(composeHangul(o, 'ㅜ', 'ㄹ').toString,
193                 lastCharString)
194 
195         // 눕다
196         case HangulChar(o: Char, 'ㅜ', 'ㅂ') =>
197           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
198               addPreEomi(composeHangul(o, 'ㅜ', ' '), PRE_EOMI_1_4 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
199               Seq(lastCharString)
200 
201         // 간지럽다, 갑작스럽다 -> 갑작스런
202         case HangulChar(o: Char, 'ㅓ', 'ㅂ') if isAdjective =>
203           addPreEomi(composeHangul(o, 'ㅓ', ' '), PRE_EOMI_1_4 ++ PRE_EOMI_7) ++
204               Seq(composeHangul(o, 'ㅓ', ' ').toString, composeHangul(o, 'ㅓ', 'ㄴ').toString, lastCharString)
205 
206         // 아름답다, 가볍다, 덥다, 간지럽다
207         case HangulChar(o: Char, v: Char, 'ㅂ') if isAdjective =>
208           addPreEomi(composeHangul(o, v, ' '), PRE_EOMI_1_4 ++ PRE_EOMI_7) ++
209               Seq(composeHangul(o, v, ' ').toString, lastCharString)
210 
211         // 놓다
212         case HangulChar(o: Char, 'ㅗ', 'ㅎ') =>
213           addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
214               CODAS_COMMON.map(composeHangul(o, 'ㅗ', _).toString) ++
215               Seq(composeHangul(o, 'ㅘ', ' ').toString, composeHangul(o, 'ㅗ', ' ').toString, lastCharString)
216 
217         // 파랗다, 퍼렇다, 어떻다
218         case HangulChar(o: Char, v: Char, 'ㅎ') if isAdjective =>
219           CODAS_COMMON.map(composeHangul(o, v, _).toString) ++
220               CODAS_FOR_CONTRACTION.map(composeHangul(o, 'ㅐ', _).toString) ++
221               Seq(composeHangul(o, 'ㅐ', ' ').toString,
222                 composeHangul(o, v, ' ').toString,
223                 lastCharString)
224 
225         // 1 char with coda adjective, 있다, 컸다
226         case HangulChar(o: Char, v: Char, c: Char) if word.length == 1 || (isAdjective && c == 'ㅆ') =>
227           addPreEomi(lastChar,
228             PRE_EOMI_COMMON ++ PRE_EOMI_1_2 ++ PRE_EOMI_1_3 ++ PRE_EOMI_2 ++ PRE_EOMI_4 ++ PRE_EOMI_5 ++ PRE_EOMI_6) ++
229               Seq(lastCharString)
230 
231         // 1 char with coda adjective, 밝다
232         case HangulChar(o: Char, v: Char, c: Char) if word.length == 1 && isAdjective =>
233           addPreEomi(lastChar,
234             PRE_EOMI_COMMON ++ PRE_EOMI_1_2 ++ PRE_EOMI_1_3 ++ PRE_EOMI_2 ++ PRE_EOMI_4 ++ PRE_EOMI_5) ++
235               Seq(lastCharString)
236 
237         // 부여잡다, 얻어맞다, 얻어먹다
238         case _ =>
239           Seq(lastCharString)
240 
241       }
242 
243       // -르 불규칙 (고르다 -> 골르다)
244       val irregularExpansion = if (lastChar == '르' && !hasCoda(init.last)) {
245         val lastInitCharDecomposed = decomposeHangul(init.last)
246         val newInit = init.init +
247           composeHangul(lastInitCharDecomposed.onset, lastInitCharDecomposed.vowel, 'ㄹ')
248 
249         val o = lastCharDecomposed.onset
250         val conjugation: Seq[String] = addPreEomi(lastChar, PRE_EOMI_2 ++ PRE_EOMI_6) ++
251           CODAS_NO_PAST.map(composeHangul(o, 'ㅡ', _).toString) ++
252           Seq(composeHangul(o, 'ㅝ').toString,
253             composeHangul(o, 'ㅓ').toString,
254             composeHangul(o, 'ㅏ').toString,
255             composeHangul(o, 'ㅝ', 'ㅆ').toString,
256             composeHangul(o, 'ㅓ', 'ㅆ').toString,
257             composeHangul(o, 'ㅏ', 'ㅆ').toString,
258             lastCharString)
259 
260         conjugation.map(newInit + _)
261       } else {
262         Seq()
263       }
264 
265       expandedLast.map(init + _) ++ irregularExpansion
266     }
267 
268     if (isAdjective) {
269       expanded
270     } else {
271       // Edge cases: these more likely to be a conjugation of an adjective than a verb
272       expanded -- Set("아니", "입", "입니", "나는")
273     }
274   }
275 }
